DROP TABLE IF EXISTS "berita";
CREATE TABLE "berita" (
  "id_berita" SERIAL PRIMARY KEY NOT NULL,
  "id_user" int NOT NULL,
  "id_kategori_berita" int NOT NULL,
  "updater" varchar(255) NOT NULL,
  "slug_berita" varchar(255) NOT NULL,
  "judul_berita" varchar(255) NOT NULL,
  "isi" text NOT NULL,
  "gambar" varchar(255) NOT NULL,
  "hits" int NOT NULL,
  "status_berita" int NOT NULL,
  "jenis_berita" varchar(20) NOT NULL,
  "keywords" varchar(500) NOT NULL,
  "tanggal_mulai" date DEFAULT NULL,
  "tanggal_selesai" date DEFAULT NULL,
  "tanggal_post" timestamp NOT NULL,
  "tanggal" timestamp NOT NULL DEFAULT (CURRENT_TIMESTAMP)
);

DROP TABLE IF EXISTS "galeri";
CREATE TABLE "galeri" (
  "id_galeri" SERIAL PRIMARY KEY NOT NULL,
  "id_user" int NOT NULL,
  "judul_galeri" varchar(255) NOT NULL,
  "isi_galeri" text NOT NULL,
  "website" varchar(255) DEFAULT NULL,
  "hits" int NOT NULL,
  "gambar" varchar(255) NOT NULL,
  "posisi_galeri" int NOT NULL,
  "tanggal_post" timestamp NOT NULL,
  "tanggal" timestamp NOT NULL DEFAULT (CURRENT_TIMESTAMP)
);

DROP TABLE IF EXISTS "kategori_berita";
CREATE TABLE "kategori_berita" (
  "id_kategori_berita" SERIAL PRIMARY KEY NOT NULL,
  "nama_kategori" varchar(255) NOT NULL,
  "slug_kategori" varchar(255) NOT NULL,
  "urutan" int NOT NULL,
  "tanggal" timestamp NOT NULL DEFAULT (CURRENT_TIMESTAMP)
);

DROP TABLE IF EXISTS "konfigurasi";
CREATE TABLE "konfigurasi" (
  "id_konfigurasi" SERIAL PRIMARY KEY NOT NULL,
  "id_user" int NOT NULL,
  "namaweb" varchar(50) NOT NULL,
  "tagline" varchar(100) DEFAULT NULL,
  "email" varchar(255) DEFAULT NULL,
  "website" varchar(255) DEFAULT NULL,
  "telepon" varchar(25) DEFAULT NULL,
  "alamat" text DEFAULT NULL,
  "deskripsi" varchar(300) DEFAULT NULL,
  "keywords" varchar(300) DEFAULT NULL,
  "metatext" text DEFAULT NULL,
  "map" text DEFAULT NULL,
  "logo" varchar(255) DEFAULT NULL,
  "icon" varchar(255) DEFAULT NULL,
  "facebook" varchar(255) DEFAULT NULL,
  "instagram" varchar(255) DEFAULT NULL,
  "tanggal" timestamp NOT NULL DEFAULT (CURRENT_TIMESTAMP)
);

DROP TABLE IF EXISTS "kontak";
CREATE TABLE "kontak" (
  "id_kontak" SERIAL PRIMARY KEY NOT NULL,
  "nama" varchar(100) NOT NULL,
  "email" varchar(255) NOT NULL,
  "subject" varchar(255) NOT NULL,
  "pesan" text NOT NULL,
  "tanggal" timestamp NOT NULL DEFAULT (CURRENT_TIMESTAMP)
);

DROP TABLE IF EXISTS "layanan";
CREATE TABLE "layanan" (
  "id_layanan" SERIAL PRIMARY KEY NOT NULL,
  "id_user" int NOT NULL,
  "hits" int NOT NULL,
  "judul_layanan" varchar(255) NOT NULL,
  "slug_layanan" varchar(255) NOT NULL,
  "isi_layanan" text NOT NULL,
  "harga" varchar(255) NOT NULL,
  "gambar" varchar(255) NOT NULL,
  "status_layanan" varchar(25) NOT NULL,
  "keywords" varchar(500) DEFAULT NULL,
  "tanggal_post" timestamp NOT NULL DEFAULT (CURRENT_TIMESTAMP),
  "tanggal" timestamp NOT NULL DEFAULT (CURRENT_TIMESTAMP)
);

DROP TABLE IF EXISTS "user";
CREATE TABLE "users" (
  "id_user" SERIAL PRIMARY KEY NOT NULL,
  "nama" varchar(50) NOT NULL,
  "email" varchar(200) NOT NULL,
  "username" varchar(255) NOT NULL,
  "password" varchar(255) NOT NULL,
  "akses_level" varchar(255) NOT NULL,
  "tanggal" timestamp NOT NULL DEFAULT (CURRENT_TIMESTAMP)
);

DROP TABLE IF EXISTS "users_token";
CREATE TABLE "users_token" (
  "id_token" SERIAL PRIMARY KEY NOT NULL,
  "email" varchar(255) NOT NULL,
  "user_token" varchar(255) NOT NULL,
  "tanggal_buat" int NOT NULL
);
